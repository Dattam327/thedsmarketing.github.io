        const closeIconSpan = mobileMenuButton.querySelector('.icon-x');


        mobileMenuButton.addEventListener('click', () => {

            const isExpanded = mobileMenuButton.getAttribute('aria-expanded') === 'true' || false;

            mobileMenuButton.setAttribute('aria-expanded', !isExpanded);

            mobileMenu.classList.toggle('hidden');

            openIconSpan.classList.toggle('hidden');

            openIconSpan.classList.toggle('block');

            closeIconSpan.classList.toggle('hidden');

            closeIconSpan.classList.toggle('block');

        });


        // Smooth scroll for anchor links

        document.querySelectorAll('a[href^="#"]').forEach(anchor => {

            anchor.addEventListener('click', function (e) {

                const targetId = this.getAttribute('href');

                if (targetId && targetId.startsWith('#') && targetId.length > 1) {

                    e.preventDefault();

                    const targetElement = document.querySelector(targetId);

                    if (targetElement) {

                        const navHeight = document.querySelector('nav').offsetHeight || 0;

                        const elementPosition = targetElement.getBoundingClientRect().top;

                        const offsetPosition = elementPosition + window.pageYOffset - navHeight;

                    

                        window.scrollTo({ top: offsetPosition, behavior: 'smooth' });

                        if (mobileMenu && !mobileMenu.classList.contains('hidden')) {

                            mobileMenuButton.click();

                        }

                    }

                }

            });

        });

        

        document.getElementById('currentYear').textContent = new Date().getFullYear();


        const generateIdeasButton = document.getElementById('generateIdeasButton');

        const aiServiceInput = document.getElementById('aiServiceInput');

        const aiIdeasResult = document.getElementById('aiIdeasResult');

        const aiLoadingSpinner = document.getElementById('aiLoadingSpinner');


        if (generateIdeasButton) {

            generateIdeasButton.addEventListener('click', async () => {

                const userService = aiServiceInput.value.trim();

                aiIdeasResult.innerHTML = ''; 

                

                if (!userService) {

                    aiIdeasResult.innerHTML = '<p class="text-red-400">Please enter a service (e.g., Oil Change) to generate ideas.</p>';

                    return;

                }


                aiLoadingSpinner.classList.remove('hidden'); 


                const prompt = `You are a helpful marketing assistant for auto repair shops. Generate 3 short, catchy marketing taglines for an auto repair shop promoting their "${userService}" service. The taglines should be aimed at attracting local customers. Each tagline should be on a new line.`;

                

                let chatHistory = [{ role: "user", parts: [{ text: prompt }] }];

                const payload = { contents: chatHistory };

                const apiKey = ""; 

                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;


                try {

                    const response = await fetch(apiUrl, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify(payload)

                    });


                    if (!response.ok) {

                        const errorData = await response.json();

                        console.error('API Error:', errorData);

                        throw new Error(`API request failed with status ${response.status}. ${errorData?.error?.message || 'Please try again.'}`);

                    }


                    const result = await response.json();

                    

                    if (result.candidates && result.candidates.length > 0 &&

                        result.candidates[0].content && result.candidates[0].content.parts &&

                        result.candidates[0].content.parts.length > 0) {

                        const text = result.candidates[0].content.parts[0].text;

                        const taglines = text.split('\n').filter(tagline => tagline.trim() !== '');

                        

                        if (taglines.length > 0) {

                            let htmlOutput = '<h4 class="text-lg font-semibold text-neutral-200 mb-3">Here are some ideas for your "' + userService + '" service:</h4><ul class="list-disc list-inside space-y-2 text-neutral-300">';

                            taglines.forEach(tagline => {

                                htmlOutput += `<li>${tagline.trim()}</li>`;

                            });

                            htmlOutput += '</ul>';

                            aiIdeasResult.innerHTML = htmlOutput;

                        } else {

                            aiIdeasResult.innerHTML = '<p class="text-neutral-400">No taglines generated. Try a different service or rephrase your input.</p>';

                        }

                    } else {

                        console.error('Unexpected API response structure:', result);

                        aiIdeasResult.innerHTML = '<p class="text-red-400">Could not parse ideas from the AI. The response might be empty or in an unexpected format. Check console for details.</p>';

                    }


                } catch (error) {

                    console.error('Error calling Gemini API:', error);

                    aiIdeasResult.innerHTML = `<p class="text-red-400">An error occurred: ${error.message}. Please check the console for more details.</p>`;

                } finally {

                    aiLoadingSpinner.classList.add('hidden'); 

                }

            });

        }

    </script>
